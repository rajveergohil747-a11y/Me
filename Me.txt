// Typing Effect
let text = "I am Rajveersinh Gohil, a year-old Class  student and passionate app developer. I love creating apps like NikoApp and designing modern user interfaces.";
let i = 0;

function typing() {
    if (i < text.length) {
        document.getElementById("typingText").innerHTML += text.charAt(i);
        i++;
        setTimeout(typing, 30);
    }
}
typing();

// Chatbot
function sendMsg() {
    let input = document.getElementById("userInput");
    let msg = input.value;
    if (!msg) return;

    let chat = document.getElementById("chatBody");

    chat.innerHTML += `<div class="msg user">${msg}</div>`;

    let reply = getReply(msg.toLowerCase());

    setTimeout(() => {
        chat.innerHTML += `<div class="msg bot">${reply}</div>`;
        chat.scrollTop = chat.scrollHeight;
    }, 500);

    input.value = "";
}

function getReply(msg) {
    if (msg.includes("who")) {
        return "Rajveer is a year-old developer and creator of NikoApp.";
    }
    if (msg.includes("niko")) {
        return "NikoApp is a messaging and social app created by Rajveer.";
    }
    if (msg.includes("project")) {
        return "Rajveer has built NikoApp and tournament systems.";
    }
    if (msg.includes("age")) {
        return "Rajveer is years old.";
    }
    return "Rajveer is a young and talented app developer.";
}
</script>
